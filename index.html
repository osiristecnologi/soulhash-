async function connectWallet() { try { let provider; // Detect Phantom if (window.solana && window.solana.isPhantom) { provider = window.solana; console.log("Phantom detected"); } // Detect Solflare else if (window.solflare && window.solflare.isSolflare) { provider = window.solflare; console.log("Solflare detected"); } else { alert("No wallet found. Please open in Phantom or Solflare browser."); return; } // Conectar wallet const res = await provider.connect(); const publicKey = res.publicKey.toString(); userWallet = publicKey; document.getElementById("wallet").innerText = "Wallet: " + publicKey; // Enviar pro backend const response = await fetch("https://soulhash-api.onrender.com/login", { method: "POST", headers: {"Content-Type": "application/json"}, body: JSON.stringify({ wallet: publicKey }) }); const data = await response.json(); document.getElementById("soulhash").innerText = "SoulHash: " + data.soulhash; } catch (err) { console.error(err); alert("Error connecting wallet"); } }
